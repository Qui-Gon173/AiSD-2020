#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

// Функция заполнения массива array длиной n случайными числами 
void ArrayRandFill(int array[], int n)
{
    srand((unsigned)time(0));
    for (int i = 0; i < n; i++)
    {
        array[i] = rand() % 10;
    }
}

//Функция вывода массива на экран
void ArrayOut(int array[], int n)
{
    for (int i = 0; i < n; i++)
    {
        cout << array[i] << " ";
    }
    cout << endl;
}
//Функция сортировки массива методом вставок
void ArraySort(int array[], int n)
{
    for (int i = 0; i < n; i++)
    {
        int k = i;
        while (k > 0 && array[k - 1] > array[k]) {
            int  tmp = array[k - 1];
            array[k - 1] = array[k];
            array[k] = tmp;
            k -= 1;
        }
    }
}
int main()
{
    setlocale(LC_ALL, "Russian");

    int* array = new int[n]; //Создание массива длиной n

    ArrayRandFill(array, n);

    cout << "Массив: ";
    ArrayOut(array, n);

    ArraySort(array, n);

    cout << "Отсортированный массив: ";

}

## Ответы на вопросы
### Алгоритм метода пузырька
Метод сортировки пузырьком состоит в том, чтобы сравнивать попарно два соседних элемента. 
Тот, который из них оказывается боле другого, перемещается на индекс больше. 
Или попросту говоря, они меняются местами, если предшествующий элемент больше последующего.
### Достоинства и недостатки метода пузырька
К достоинствам данного метода можно отнести то, что при написание алгоритма не требуется 
дополнительных массивов.
К недостаткам можно отнести медлительность. Это один из самых длительных методов сортировки,
так как время алгоритма пропорционально квадрату количества элементов.
### Алгоритм шейкер-сортировки
Данный метод сортировки - это более модифицированный метод сортировки пузырьком. 
Отличается он тем, что та часть массива, в котором перестановки элементов уже не происходят, 
считается остсортированной, поэтому она исключается из рассмотрения и не обрабатывется далее.
Также при данном методе самый "легкий" элемент перемещается сразу "наверх" массива, а "тяжелые" 
элементы смещаются по-немногу "вниз".
### Достоинства и недостатки шейкер-сортировки
Достоинства - легкая реализация алгоритма, довольно быстрее, чем сортировка "пузырьком".
Недостатки - все также достаточно медленная работа алгоритма.
### Алгоритм метода перебора
Метод перебора заключается в том, что создается отсортированная последовательность путем
присоединения к ней одного элемента за другим в правильном порядке
### Достоинства и недостатки метода перебора
Достоинства - легкая реализация алгоритма.
Недостатки - большое количество сравнений, следовательно долгая работа алгоритма.
### Алгоритм метода вставок
Суть данного метода в том, чтоперебираются элементы в неотсортированной части массива.
Каждый элемент вставляется в отсортированную часть массива на то место, где он должен находиться.
сортировки вставками всегда делят массив на 2 части — отсортированную и неотсортированную. 
Из неотсортированной части извлекается любой элемент. Так как другая часть массива отсортирована,
то в ней быстро можно найти место для извлечённого элемента. Элемент
вставляется куда нужно, в результате чего отсортированная часть массива увеличивается, 
а неотсортированная уменьшается.
### Достоинства и недостатки метода вставок
Достатки - эффективен для массивов с небольшим количеством элементов. Также эффективен, если массив
частично отсортирован.
Недостаток - высокий уровень сложности алгоритма. Занимает много времени.
